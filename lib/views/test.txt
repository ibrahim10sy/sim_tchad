  Future<void> _fetchDataLocal() async {
    if (!mounted) return; 
    setState(() => isLoading = true);

    try {
      final dataMarche = await DatabaseService.getAllFicheMarche();
      final dataMagasin = await DatabaseService.getAllFicheMagasin();
      final dataSuivi = await DatabaseService.getAllFicheSuivi();

      final dataMarcheReject = await DatabaseService.getAllPrixMarches();
      final dataMagasinReject = await DatabaseService.getAllPrixMagasins();
      final dataSuiviReject = await DatabaseService.getAllSuivi();

      if (!mounted) return; 
     setState(() {
  fiches = dataMarche;
  fichesMagasin = dataMagasin;
  fichesSuivi = dataSuivi;

  fichesMarcheReject = dataMarcheReject;
  fichesMagasinReject = dataMagasinReject;
  fichesSuiviReject = dataSuiviReject;

  _updateTabController(); // 🔹 Met à jour le TabController dynamiquement
});
    } catch (e) {
      debugPrint("Erreur locale : $e");
    } finally {
      if (!mounted) return; 
      setState(() => isLoading = false);
    }
  }

void _updateTabController() {
  final tabs = _buildTabs();
  final newLength = tabs.length;

  if (newLength == 0) return;

  // Sauvegarde l'index actuel
  int newIndex = _tabController.index;

  if (newIndex >= newLength) {
    newIndex = newLength - 1;
  }

  // Dispose l'ancien controller si déjà initialisé
  _tabController.dispose();

  // Crée un nouveau TabController avec la bonne longueur
  _tabController = TabController(
    length: newLength,
    vsync: this,
    initialIndex: newIndex,
  );

  // Force le rebuild
  setState(() {});
}

// ---------------- BUILD TABS ----------------
 List<Tab> _buildTabs() {
  final tabs = <Tab>[
    const Tab(text: "Affectées"),
  ];

  if (hasAchevees) {
    tabs.add(
      _buildTabWithBadge(
        "Achevées",
        fiches.length + fichesMagasin.length + fichesSuivi.length,
      ),
    );
  }

  if (hasRejetees) {
    tabs.add(
      _buildTabWithBadge(
        "Rejetées",
        fichesMarcheReject.length +
            fichesMagasinReject.length +
            fichesSuiviReject.length,
      ),
    );
  }

  return tabs;
}

// ---------------- BUILD TAB VIEWS ----------------
  List<Widget> _buildTabViews() {
  final views = <Widget>[
    AffectPage(), // plus de const ici
  ];

  if (hasAchevees) {
    views.add(AchevePage());
  }

  if (hasRejetees) {
    views.add(RejetPage());
  }

  return views;
}

Future<void> _fetchDataLocal() async {
    if (!mounted) return;
    setState(() => isLoading = true);

    try {
      // Récupération de vos données
      final dataMarche = await DatabaseService.getAllFicheMarche();
      final dataMagasin = await DatabaseService.getAllFicheMagasin();
      final dataSuivi = await DatabaseService.getAllFicheSuivi();

      final dataMarcheReject = await DatabaseService.getAllPrixMarches();
      final dataMagasinReject = await DatabaseService.getAllPrixMagasins();
      final dataSuiviReject = await DatabaseService.getAllSuivi();

      if (!mounted) return;

      setState(() {
        fiches = dataMarche;
        fichesMagasin = dataMagasin;
        fichesSuivi = dataSuivi;

        fichesMarcheReject = dataMarcheReject;
        fichesMagasinReject = dataMagasinReject;
        fichesSuiviReject = dataSuiviReject;
        isLoading = false;
      });
    } catch (e) {
      debugPrint("Erreur lors de la récupération : $e");
      if (mounted) setState(() => isLoading = false);
    }
  }

  // ---------------- TABS UI ----------------
   List<Tab> _buildTabs() {
    final tabs = <Tab>[
      const Tab(text: "Affectées"),
    ];

    if (hasAchevees) {
      tabs.add(
        _buildTabWithBadge(
          "Achevées",
          fiches.length + fichesMagasin.length + fichesSuivi.length,
        ),
      );
    }

    if (hasRejetees) {
      tabs.add(
        _buildTabWithBadge(
          "Rejetées",
          fichesMarcheReject.length +
              fichesMagasinReject.length +
              fichesSuiviReject.length,
        ),
      );
    }
    return tabs;
  }

  // ---------------- CONSTRUCTION DES VUES ----------------
  List<Widget> _buildTabViews() {
    final views = <Widget>[
      const AffectPage(), // Page par défaut

    if (hasAchevees) {
      views.add(const AchevePage());

    if (hasRejetees) {
      views.add(const RejetPage());
    }
    return views;
  }

      isLoadData = true;
      syncProgress = 0;

    try {

        enqueteur!.codeEnqueteur!,
        enqueteur!.acteur!.codeActeur,
        (progress) {
          setState(() {
            syncProgress = progress;
          });
        },
      );
    } catch (e) {
      print("Erreur sync $e");
    } finally {
      setState(() {
        isLoadData = false;
      });
    }
  }
